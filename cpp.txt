sudo dnf update
# Install compilers (GCC and Clang), debugger (GDB), and build automation (Make/CMake)
sudo dnf install gcc-c++ clang clang-tools-extra cmake make ninja-build gdb valgrind

clangd --version # Language Server Protocol (LSP)

# Force Zed to use your system's clangd and enable superior auto-completion
{
  "lsp": {
    "clangd": {
      "binary": {
        "path": "/usr/bin/clangd"
      },
      "initialization_options": {
        "fallbackFlags": ["-std=c++20"]
      }
    }
  },
  "languages": {
    "C++": {
      "format_on_save": "on",
      "formatter": "language_server"
    }
  }
}

cmake -DCMAKE_EXPORT_COMPILE_COMMANDS=1 -B build .
ln -s build/compile_commands.json . # Zed will light up with perfect error checking

LLDB (debugger)
